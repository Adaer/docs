                          # vue响应式数据原理

                          vue响应式系统核心三点：

                          1. observer：遍历data中的属性，使用Object.defineProperty对属性的getters、setter方法进行数据劫持。
                          2. dep：每个属性拥有自己的消息发布订阅器dep，该属性存放在观察者对象watcher中。
                          3. watcher：观察者对象，通过dep实现对响应属性的监听，监听到结果后（此处有疑问！），主动触发回调函数进行响应，更新vnode重新渲染视图。

                          数据劫持与发布订阅

                          observer的实现与监听：

                          function observer(obj, vm){
    Object.keys(obj).forEach(function(item){
        defineReactive(vm, item, obj[item])
    })
}
function defineReactive(vm, key, val){
    // 每个属性建立单独的依赖收集对象，get中收集依赖,set中触发依赖，调用更新函数 
    var dep = new Dep();
    Object.defineProperty(vm, key, {
        enumerable: true,
        configurable: true,
        get: function(){
            //收集依赖 创建标志
            Dep.target && dep.addSub(Dep.target);
            return val
        },
        set: function(newVal){
            if(newVal === val) return
            //触发依赖 通知
            dep.notify();
            val = newVal
        }
    })
}

                          dep消息发布订阅器：

                          function Dep(){
    this.subs = []
}
Dep.prototype = {
    addSub: function(sub){
        this.subs.push(sub)
    },
    notify: function(){
        this.subs.forEach(function(sub){
            // 调用的Watcher的update方法
            sub.updata()
        })
    }
}
Object.defineProperty(Dep, 'constructor', {
    value: Dep
})

                          Watcher连接observer和compiler,接受每个属性变动的通知，绑定更新函数，更新视图

                          function Watcher(vm, node, name, nodeType){
    Dep.target = this;
    this.name = name;
    this.node = node;
    this.vm = vm;
    this.nodeType = nodeType;
    this.update(); //绑定更新函数
    Dep.target = null //绑定完成后销毁
}
Watcher.prototype = {
    get: function(){
        this.value = this.vm[this.name];
        return true
    },
    update: function(){
        this.get();
        if(this.nodeType === "text"){
            this.node.nodeValue = this.value
        }
        if(this.nodeType === "input"){
            this.node.value = this.value
        }
    }
}
Object.defineProperty(Watcher, 'constructor', {
    value: Watcher
})
