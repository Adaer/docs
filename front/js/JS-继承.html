                          # call apply bind new 的实现

                          JS的继承：

                          • 构造函数中，this上来挂载属性，prototype用来保存方法
                          • 属性继承： call apply bind
                          • 原型继承 S.prototype = F.prototype
                          • 实例继承 S.prototype = new F

                          WARNING

                          当替换了原型对象后，下一步必须将 S.prototype.constructor = S; 修正S 的 constructor 指向它的原函数

                          # call的实现

                          Function.prototype.mycall = function(target){
    //使用 Object 包装target
    var context = Object(target) || window;
    context.fn = this;

    var args = [],
        res = null;
    //参数从第二个开始
    for(var i = 1; i < arguments.length; i++){
        args.push("arguments[" + i + "]")
        //使用解构
        // args.push(arguments[i])
    }

    res = eval("context.fn("+ args +")");
    //使用解构
    //res = context.fn(...args);
    delete context.fn;
    return res
}
                          # demo
                          var t = {
  a: 9
};

function test(){
   console.log(this.a)
}
undefined
test.mycall(t)
//9

                          # apply的实现

                          Function.prototype.myApply = function(target, arr){
    var context = Object(target) || window,
        res = null,
        opt = {},
        args = [];
    
    context.fn = this;

    if(opt.toString.call(arr) !== "[object Array]"){
        res = context.fn()
    }
    else{
        //解构
        // res = context.fn(...arr)
        for(var i = 0; i < arr.length; i++){
            args.push("arr[" + i + "]")
        }

        res = eval("context.fn(" + args + ")")
    }

    return res
}
