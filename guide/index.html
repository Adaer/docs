# vuepress搭建时的问题

# 当使用 vuepress 全局安装的情况下，依赖的官方插件无法生效

module.exports = {
    plugins: [
        '@vuepress/back-to-top'
    ]    
}

这样无法使用其官方插件 需要在项目中依赖 vuepress

yarn add -D vuepress
# 或
npm install -D vuepress

# / 根目录问题

静态资源（图标/图片等）'/'默认指向的是 docs/.vuepress/public （public文件夹需手动创建）

module.exports = {
    base: '/docs/',// '/'
}

所以当这里的根目录发生变化时

![An image](/docs/avatar.jpg) 正确 
![An image](/avatar.jpg) 路径错误
![An image](./avatar.jpg) 使用相对路径

# 设置meta标签

module.exports = {
    head: [
        ['link', { rel: 'icon', href: '/favicon.ico' }],
        ['meta', { charset: 'UTF-8'}],
        ['meta', { name: 'renderer', content: 'webkit'}],
        ['meta', { 'http-equiv': 'X-UA-Compatible', content: 'IE=edge, chrome=1'}]
    ],
}
